## 3. 练习2参考答案:LeafletMap组件

### LeafletMap.vue

```vue
<template>
  <div class="leaflet-map-container">
    <div 
      ref="mapRef" 
      :style="{ height: height }"
      class="map-element"
    />
    <div v-if="loading" class="loading-overlay">
      <div class="spinner" />
      <p>地图加载中...</p>
    </div>
  </div>
</template>

<script setup lang="ts">
import { onMounted, onUnmounted, ref, watch } from 'vue';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';

/**
 * Leaflet地图组件Props
 */
interface Props {
  /** 地图中心坐标[经度, 纬度] */
  center?: [number, number];
  /** 缩放级别 */
  zoom?: number;
  /** 地图容器高度 */
  height?: string;
}

/**
 * 组件事件
 */
interface Emits {
  /** 地图点击事件 */
  (e: 'map-click', coords: { lat: number; lng: number }): void;
  /** 地图准备就绪 */
  (e: 'map-ready', map: L.Map): void;
}

// Props with defaults
const props = withDefaults(defineProps<Props>(), {
  center: () => [116.4, 39.9],
  zoom: 10,
  height: '500px'
});

const emit = defineEmits<Emits>();

// Refs
const mapRef = ref<HTMLDivElement | null>(null);
const loading = ref(true);
let map: L.Map | null = null;
const markers: L.Marker[] = [];

/**
 * 初始化地图
 */
const initMap = () => {
  if (!mapRef.value) {
    console.error('Map container not found');
    return;
  }

  try {
    // 创建地图实例
    map = L.map(mapRef.value, {
      center: [props.center[1], props.center[0]], // Leaflet使用[lat, lng]顺序
      zoom: props.zoom,
      zoomControl: true
    });

    // 添加OpenStreetMap底图
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors',
      maxZoom: 19
    }).addTo(map);

    // 监听地图点击事件
    map.on('click', (e: L.LeafletMouseEvent) => {
      const { lat, lng } = e.latlng;
      console.log(`地图点击坐标: [${lng}, ${lat}]`);
      emit('map-click', { lat, lng });
    });

    // 地图加载完成
    map.whenReady(() => {
      loading.value = false;
      emit('map-ready', map!);
    });

  } catch (error) {
    console.error('地图初始化失败:', error);
    loading.value = false;
  }
};

/**
 * 添加标注点
 * @param lng - 经度
 * @param lat - 纬度
 * @param popupContent - 弹窗内容
 */
const addMarker = (lng: number, lat: number, popupContent: string = '') => {
  if (!map) {
    console.warn('地图未初始化');
    return null;
  }

  const marker = L.marker([lat, lng]).addTo(map);
  
  if (popupContent) {
    marker.bindPopup(popupContent);
  }
  
  markers.push(marker);
  return marker;
};

/**
 * 清除所有标注点
 */
const clearMarkers = () => {
  markers.forEach(marker => marker.remove());
  markers.length = 0;
};

/**
 * 设置地图视图
 * @param lng - 经度
 * @param lat - 纬度  
 * @param zoom - 缩放级别
 */
const setView = (lng: number, lat: number, zoom?: number) => {
  if (!map) return;
  map.setView([lat, lng], zoom || map.getZoom());
};

// 监听center变化
watch(() => props.center, (newCenter) => {
  if (map) {
    map.setView([newCenter[1], newCenter[0]], map.getZoom());
  }
});

// 监听zoom变化
watch(() => props.zoom, (newZoom) => {
  if (map) {
    map.setZoom(newZoom);
  }
});

// 生命周期
onMounted(() => {
  initMap();
});

onUnmounted(() => {
  if (map) {
    map.remove();
    map = null;
  }
});

// 暴露方法
defineExpose({
  addMarker,
  clearMarkers,
  setView,
  getMap: () => map
});
</script>

<style scoped>
.leaflet-map-container {
  position: relative;
  width: 100%;
  border-radius: 8px;
  overflow: hidden;
}

.map-element {
  width: 100%;
}

.loading-overlay {
  position: absolute;
  top: 0;
  left: 0;
  right: 0;
  bottom: 0;
  background: rgba(255, 255, 255, 0.9);
  display: flex;
  flex-direction: column;
  align-items: center;
  justify-content: center;
  z-index: 1000;
}

.spinner {
  width: 50px;
  height: 50px;
  border: 4px solid #f3f3f3;
  border-top: 4px solid #3498db;
  border-radius: 50%;

}

@keyframes spin {
  0% { transform: rotate(0deg); }
  100% { transform: rotate(360deg); }
}
```

### 使用示例

```vue
<template>
  <div class="app">
    <h1>Leaflet地图示例</h1>
    
    <LeafletMap
      :center="mapCenter"
      :zoom="mapZoom"
      height="600px"
      @map-click="handleMapClick"
      @map-ready="handleMapReady"
      ref="mapComponent"
    />
    
    <div class="controls">
      <button @click="addSampleMarker">添加示例标注</button>
      <button @click="clearAllMarkers">清除标注</button>
      <button @click="flyToBeijing">飞到北京</button>
    </div>
  </div>

<script setup lang="ts">
import { ref } from 'vue';
import LeafletMap from './components/LeafletMap.vue';

const mapCenter = ref<[number, number]>([116.4, 39.9]);
const mapZoom = ref(10);
const mapComponent = ref<InstanceType<typeof LeafletMap> | null>(null);

const handleMapClick = (coords: { lat: number; lng: number }) => {
  console.log('地图被点击:', coords);
  // 在点击位置添加标注
  mapComponent.value?.addMarker(
    coords.lng, 
    coords.lat, 
    `坐标: ${coords.lat.toFixed(4)}, ${coords.lng.toFixed(4)}`
  );
};

const handleMapReady = (map: L.Map) => {
  console.log('地图加载完成', map);
};

const addSampleMarker = () => {
  mapComponent.value?.addMarker(
    116.4074, 
    39.9042, 
    '<b>天安门</b><br>北京市中心'
  );
};

const clearAllMarkers = () => {
  mapComponent.value?.clearMarkers();
};

const flyToBeijing = () => {
  mapComponent.value?.setView(116.4, 39.9, 12);
};
</script>
```

```typescript
  const error = ref<Error | null>(null);
  
  /**
   * 加载POI数据
   */
  const loadPOIs = async () => {
    loading.value = true;
    error.value = null;
    
    try {
      const response = await fetch('/src/data/pois.json');
      if (!response.ok) {
        throw new Error('数据加载失败');
      }
      allPOIs.value = await response.json();
    } catch (err) {
      error.value = err as Error;
      console.error('加载POI数据失败:', err);
    } finally {
      loading.value = false;
    }
  };
  
  /**
   * 筛选后的POI列表
   */
  const filteredPOIs = computed(() => {
    let result = allPOIs.value;
    
    // 类别筛选
    if (filterOptions.value.selectedCategories.length > 0) {
      result = result.filter(poi => 
        filterOptions.value.selectedCategories.includes(poi.type)
      );
    }
    
    // 关键词搜索
    const keyword = filterOptions.value.searchKeyword.toLowerCase().trim();
    if (keyword) {
      result = result.filter(poi =>
        poi.name.toLowerCase().includes(keyword) ||
        poi.description.toLowerCase().includes(keyword)
      );
    }
    
    return result;
  });
  
  /**
   * 设置类别筛选
   */
  const setCategoryFilter = (categories: POICategory[]) => {
    filterOptions.value.selectedCategories = categories;
  };
  
  /**
   * 设置搜索关键词
   */
  const setSearchKeyword = (keyword: string) => {
    filterOptions.value.searchKeyword = keyword;
  };
  
  /**
   * 清除所有筛选
   */
  const clearFilters = () => {
    filterOptions.value.selectedCategories = [];
    filterOptions.value.searchKeyword = '';
  };
  
  /**
   * 根据ID查找POI
   */
  const findPOIById = (id: number): POI | undefined => {
    return allPOIs.value.find(poi => poi.id === id);
  };
  
  return {
    // 状态
    allPOIs,
    filteredPOIs,
    loading,
    error,
    filterOptions,
    
    // 方法
    loadPOIs,
    setCategoryFilter,
    setSearchKeyword,
    clearFilters,
    findPOIById
  };
}
```
