## 4. Windows 环境变量配置脚本（PowerShell）

保存为 `setup-env.ps1`，以管理员身份运行：

```powershell
# GIS开发环境变量配置脚本
# 运行方式：以管理员身份打开PowerShell，执行 .\setup-env.ps1

Write-Host "========================================" -ForegroundColor Cyan
Write-Host "  GIS开发环境变量配置脚本" -ForegroundColor Cyan
Write-Host "========================================" -ForegroundColor Cyan
Write-Host ""

# 检查是否以管理员身份运行
$isAdmin = ([Security.Principal.WindowsPrincipal] [Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole]::Administrator)
if (-not $isAdmin) {
    Write-Host "错误：请以管理员身份运行此脚本！" -ForegroundColor Red
    exit 1
}

# 设置执行策略
Write-Host "设置PowerShell执行策略..." -ForegroundColor Yellow
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser -Force

# 配置JAVA_HOME（需要根据实际安装路径修改）
$javaPath = "C:\Program Files\Eclipse Adoptium\jdk-21.0.2+13"
if (Test-Path $javaPath) {
    Write-Host "配置JAVA_HOME..." -ForegroundColor Yellow
    [Environment]::SetEnvironmentVariable("JAVA_HOME", $javaPath, "Machine")
    
    # 添加到PATH
    $currentPath = [Environment]::GetEnvironmentVariable("Path", "Machine")
    if (-not $currentPath.Contains("%JAVA_HOME%\bin")) {
        $newPath = $currentPath + ";%JAVA_HOME%\bin"
        [Environment]::SetEnvironmentVariable("Path", $newPath, "Machine")
        Write-Host "JAVA_HOME已添加到PATH" -ForegroundColor Green
    }
} else {
    Write-Host "警告：未找到JDK安装目录，请手动配置JAVA_HOME" -ForegroundColor Yellow
    Write-Host "预期路径：$javaPath" -ForegroundColor Yellow
}

# 配置npm镜像
Write-Host "配置npm淘宝镜像..." -ForegroundColor Yellow
npm config set registry https://registry.npmmirror.com

# 配置pip镜像
Write-Host "配置pip清华镜像..." -ForegroundColor Yellow

Write-Host "========================================" -ForegroundColor Green
```

## 5. macOS/Linux 环境配置脚本（Bash）

保存为 `setup-env.sh`：

```bash
#!/bin/bash

# GIS开发环境配置脚本（macOS/Linux）
# 运行方式：chmod +x setup-env.sh && ./setup-env.sh

echo "========================================"
echo "  GIS开发环境配置脚本"
echo "========================================"
echo ""

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 检测操作系统
if [[ "$OSTYPE" == "darwin"* ]]; then
    OS="macOS"
    SHELL_RC="$HOME/.zshrc"
elif [[ "$OSTYPE" == "linux-gnu"* ]]; then
    OS="Linux"
    SHELL_RC="$HOME/.bashrc"
else
    echo -e "${RED}不支持的操作系统：$OSTYPE${NC}"
    exit 1
fi

echo -e "${YELLOW}检测到操作系统：$OS${NC}"
echo ""

# macOS：检查并安装Homebrew
if [[ "$OS" == "macOS" ]]; then
    if ! command -v brew &> /dev/null; then
        echo -e "${YELLOW}安装Homebrew...${NC}"
        /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
    else
        echo -e "${GREEN}Homebrew已安装${NC}"
    fi
fi

# 安装Node.js
echo ""
echo -e "${YELLOW}检查Node.js...${NC}"
if ! command -v node &> /dev/null; then
    echo "安装Node.js 22..."
    if [[ "$OS" == "macOS" ]]; then
        brew install node@22
    else
        # Linux使用NodeSource
        curl -fsSL https://deb.nodesource.com/setup_22.x | sudo -E bash -
        sudo apt-get install -y nodejs
    fi
else
    echo -e "${GREEN}Node.js已安装：$(node --version)${NC}"
fi

# 安装Python
echo ""
echo -e "${YELLOW}检查Python...${NC}"
if ! command -v python3 &> /dev/null; then
    echo "安装Python 3.12..."
    if [[ "$OS" == "macOS" ]]; then
        brew install python@3.12
    else
        sudo apt-get update
        sudo apt-get install -y python3.12 python3.12-venv python3-pip
    fi
else
    echo -e "${GREEN}Python已安装：$(python3 --version)${NC}"
fi

# 安装JDK
echo ""
echo -e "${YELLOW}检查JDK...${NC}"
if ! command -v java &> /dev/null; then
    echo "安装JDK 21..."
    if [[ "$OS" == "macOS" ]]; then
        brew install --cask temurin@21
    else
        sudo apt-get install -y temurin-21-jdk
    fi
else
    echo -e "${GREEN}JDK已安装：$(java --version 2>&1 | head -1)${NC}"
fi

# 配置npm镜像
echo ""
echo -e "${YELLOW}配置npm镜像...${NC}"
npm config set registry https://registry.npmmirror.com

# 配置pip镜像
echo ""
echo -e "${YELLOW}配置pip镜像...${NC}"
pip3 config set global.index-url https://pypi.tuna.tsinghua.edu.cn/simple

# 安装pnpm
echo ""
echo -e "${YELLOW}安装pnpm...${NC}"
npm install -g pnpm

# 配置JAVA_HOME（添加到shell配置文件）
echo ""
echo -e "${YELLOW}配置环境变量...${NC}"

# 检查JAVA_HOME是否已配置
if ! grep -q "JAVA_HOME" "$SHELL_RC"; then
    if [[ "$OS" == "macOS" ]]; then
        JAVA_HOME_PATH="/Library/Java/JavaVirtualMachines/temurin-21.jdk/Contents/Home"
    else
        JAVA_HOME_PATH="/usr/lib/jvm/temurin-21-jdk-amd64"
    fi
    
    echo "" >> "$SHELL_RC"
    echo "# Java环境变量" >> "$SHELL_RC"
    echo "export JAVA_HOME=\"$JAVA_HOME_PATH\"" >> "$SHELL_RC"
    echo "export PATH=\"\$JAVA_HOME/bin:\$PATH\"" >> "$SHELL_RC"
    
    echo -e "${GREEN}JAVA_HOME已添加到 $SHELL_RC${NC}"
fi

echo ""
echo "========================================"
echo -e "${GREEN}  配置完成！${NC}"
echo "========================================"
echo ""
echo "请运行以下命令使配置生效："
echo -e "${YELLOW}  source $SHELL_RC${NC}"
echo ""
echo "或者重新打开终端窗口"
```

## 6. 环境验证脚本（Python）

保存为 `verify_env.py`：

```python
#!/usr/bin/env python3
"""
GIS开发环境验证脚本
运行方式：python verify_env.py
"""

import subprocess
import sys
import shutil
from typing import Tuple, Optional


def run_command(command: str) -> Tuple[bool, str]:
    """
    运行命令并返回结果
    
    Args:
        command: 要执行的命令
        
    Returns:
        (是否成功, 输出内容)
    """
    try:
        result = subprocess.run(
            command,
            shell=True,
            capture_output=True,
            text=True,
            timeout=30
        )
        if result.returncode == 0:
            return True, result.stdout.strip()
        else:
            return False, result.stderr.strip()
    except subprocess.TimeoutExpired:
        return False, "命令执行超时"
    except Exception as e:
        return False, str(e)


def check_command_exists(command: str) -> bool:
    """检查命令是否存在"""
    return shutil.which(command) is not None


def print_result(name: str, success: bool, version: str = "", required: str = ""):
    """打印检查结果"""
    status = "✅" if success else "❌"
    if success:
        print(f"  {status} {name}: {version}")
    else:
        print(f"  {status} {name}: 未安装或配置错误")
        if required:
            print(f"      要求版本: {required}")


def main():
    """主函数"""
    print("=" * 50)
    print("  GIS开发环境验证")
    print("=" * 50)
    print()
    
    results = []
    
    # 检查Node.js
    print("📦 Node.js 环境")
    print("-" * 30)
    
    success, output = run_command("node --version")
    print_result("Node.js", success, output, "v22.x.x")
    results.append(("Node.js", success))
    
    success, output = run_command("npm --version")
    print_result("npm", success, output, "10.x.x")
    results.append(("npm", success))
    
    success, output = run_command("pnpm --version")
    print_result("pnpm", success, output, "(可选)")
    
    print()
    
    # 检查Python
    print("🐍 Python 环境")
    print("-" * 30)
    
    # 尝试python3和python
    python_cmd = "python3" if check_command_exists("python3") else "python"
    success, output = run_command(f"{python_cmd} --version")
    print_result("Python", success, output, "3.12+")
    results.append(("Python", success))
    
    pip_cmd = "pip3" if check_command_exists("pip3") else "pip"
    success, output = run_command(f"{pip_cmd} --version")
    if success:
        # 只取版本号部分
        version = output.split()[1] if output else ""
        print_result("pip", success, f"pip {version}")
    else:
        print_result("pip", success)
    results.append(("pip", success))
    
    print()
    
    # 检查Java
    print("☕ Java 环境")
    print("-" * 30)
    
    success, output = run_command("java --version")
    if success:
        # 取第一行
        version = output.split('\n')[0] if output else ""
        print_result("Java", success, version, "21+")
    else:
        print_result("Java", success, required="21+")
    results.append(("Java", success))
    
    # 检查JAVA_HOME
    import os
    java_home = os.environ.get("JAVA_HOME", "")
    if java_home:
        print(f"  ✅ JAVA_HOME: {java_home}")
    else:
        print("  ⚠️  JAVA_HOME: 未设置（部分工具可能无法正常工作）")
    
    print()
    
    # 检查Git
    print("🔧 开发工具")
    print("-" * 30)
    
    success, output = run_command("git --version")
    if success:
        version = output.replace("git version ", "")
        print_result("Git", success, version)
    else:
        print_result("Git", success)
    results.append(("Git", success))
    
    # 检查Docker
    success, output = run_command("docker --version")
    if success:
        version = output.split()[2].rstrip(',') if output else ""
        print_result("Docker", success, version, "(可选)")
    else:
        print_result("Docker", success, required="(可选)")
    
    print()
    
    # 汇总结果
    print("=" * 50)
    print("  检查结果汇总")
    print("=" * 50)
    
    total = len(results)
    passed = sum(1 for _, success in results if success)
    
    if passed == total:
        print(f"\n🎉 太棒了！所有必要组件都已正确安装 ({passed}/{total})")
        print("\n您的开发环境已准备就绪，可以开始GIS开发之旅了！")
    else:
        print(f"\n⚠️  部分组件未安装 ({passed}/{total})")
        print("\n请安装以下缺失的组件：")
        for name, success in results:
            if not success:
                print(f"  - {name}")
        print("\n安装完成后，请重新运行此脚本验证。")
    
    print()
    
    return 0 if passed == total else 1


if __name__ == "__main__":
    sys.exit(main())
```
